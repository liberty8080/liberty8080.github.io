### 重命名脚本

```python
import os


	
def my_rename(path):
	# 遍历文件夹 
	for file in os.listdir(path):
		old_name = os.path.join(path,file)
		# print("旧文件名为："+old_name)
		new_name = file.replace("哔哩哔哩-ssm框架角色权限后台管理系统脚手架开发实战教程-","")
		print("新文件名为:  "+new_name)
		# 生成新文件名
		os.rename(old_name,os.path.join(path,new_name))


#递归版
def math_rename(root_path):
	for file in os.listdir(root_path):
		path = os.path.join(root_path, file)
		if os.path.isdir(path):
			math_rename(path)
		else:
			old_name = path
			new_name = file.replace("19考研数学基础班","")
			new_name = new_name.replace("高等数学汤家凤加密","")
			new_name = new_name.replace("高等数学汤老师加密","")
			print('新文件名为：  '+new_name)
			os.rename(old_name,os.path.join(path,new_name))


root_path = "E:\\05.高等数学基础【完】"
math_rename(root_path)
```
